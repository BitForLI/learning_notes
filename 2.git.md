#常见命令
git config  git config --global --list查看和修改配置
git init  将一个文件初始化成git仓库
git clone  从远程仓库复制到本地
git add . 将文件加入到暂存区
git reset  回退
git commit -m  “”   提交到本地历史
git status  查看文件跟踪状态
git push  本地分支提交到远程仓库
git pull  从远程拉取
git branch  查看/新建/切换分支
git merge  合并分支
git log  查看提交历史
git stash  隐藏未提交改动
git rm  删除文件
git remote -v 查看远程仓库地址

标准流程
# 1. 第一次拉代码
git clone https://github.com/user/project.git
cd project

# 2. 配置用户名邮箱（只需第一次）
git config user.name "Your Name"
git config user.email "you@example.com"

# 3. 更新主分支
git checkout main
git pull

# 4. 新建功能分支
git checkout -b feature/login

# 5. 写代码，改文件...
git status          # 看改了什么
git add src/login.js
git commit -m "feat: add login page"
git log --oneline   # 看一下提交历史

# 中途突然要切到别的分支修 bug，但现在代码没写完：
git stash           # 把当前改动藏起来
git checkout main
git checkout -b hotfix/xxx
# 修复、提交、push、合并完以后
git checkout feature/login
git stash pop       # 把之前的改动拿回来继续写

# 6. 功能完成，推送远程
git push -u origin feature/login

# 7. 在远程（GitHub/GitLab）发起 Merge Request / Pull Request
# 审核通过后合并到 main

# 8. 本地 main 更新
git checkout main
git pull
