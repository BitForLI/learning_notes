# 常见命令
*查看帮助*
docker --version         # 看版本
docker info              # 看整体状态：镜像数、容器数、驱动等
docker help              # 总帮助
docker <子命令> --help   # 某条命令的用法，例如: docker run --help

*镜像相关*
**查看、搜索、拉取**
docker images            # 列出本地所有镜像
docker images nginx      # 只列出名字里包含 nginx 的镜像
docker search nginx      # 在 Docker Hub 搜索 nginx 镜像
docker pull nginx:latest # 从远程仓库拉取 nginx:latest
docker pull ubuntu:20.04
**构建**
docker build -t myapp:1.0 .
**打标签、删除**
docker tag nginx:latest exlli/mynginx:latest   # 重命名/打新标签
docker rmi nginx:latest                       # 删除镜像
docker rmi exlli/mynginx:latest nginx:v1.1    # 一次删多个

*容器：创建、启动、停止*
**运行容器**
docker run nginx:latest     # 前台运行，看日志输出
docker run -d --name mynginx -p 8080:80 nginx:latest     # 后台运行 + 指定容器名 + 端口映射
docker run --rm alpine:latest echo "hello"  # 运行一次性命令，用完即退出

-d：后台运行（detached）
--name：容器名字
-p 宿主端口:容器端口：端口映射
--rm：容器退出后自动删除
**查看容器**
docker ps           # 正在运行的容器
docker ps -a        # 所有容器（包括已退出的）
**停止、启动、重启、删除容器**
docker stop mynginx           # 停止
docker start mynginx          # 再次启动
docker restart mynginx        # 重启
docker rm mynginx             # 删除一个已停止的容器
docker rm -f mynginx          # 强制删除（如果在运行会先杀掉）
docker container prune  # 一次删掉所有已停止容器

*查看日志、进入容器*
**查看容器日志**
docker logs mynginx           # 打印日志
docker logs -f mynginx        # 实时跟随日志（类似 tail -f）
docker logs --tail 100 mynginx  # 只看最后 100 行
**进入容器执行命令**
docker exec -it mynginx /bin/bash   # 交互式进入容器，打开 bash
docker exec mynginx nginx -v    # 在容器中执行一次命令
docker exec -it some-container /bin/sh  # 没有 bash 的轻量镜像，用 sh

*文件、数据卷*
**宿主机和容器之间拷贝文件**
docker cp mynginx:/etc/nginx/nginx.conf .       # 从容器拷贝到宿主机
docker cp ./index.html mynginx:/usr/share/nginx/html/       # 从宿主机拷贝到容器
**使用数据卷**
docker volume create mydata     # 创建数据卷

docker run -d --name mymysql \
  -e MYSQL_ROOT_PASSWORD=123456 \
  -v mydata:/var/lib/mysql \
  mysql:8                               # 在运行容器时挂载数据卷

docker volume ls       # 列出数据卷
docker volume rm mydata
docker volume prune    # 删除未被使用的数据卷

*网络相关*
docker network ls                         # 列出网络
docker network inspect bridge             # 查看某个网络详细信息
docker network create mynet               # 创建一个自定义网络
docker run -d --name web --network=mynet nginx:latest       # 运行容器时接入某个网络

*清理资源*
docker system df                # 查看镜像/容器/卷占用空间
docker system prune             # 清理无用的容器、网络、缓存（会问你确认）
docker system prune -a          # 额外删除所有“悬空镜像”（谨慎）

*与远程仓库交互*
docker login                    # 登录 Docker Hub
docker logout                   # 登出

docker tag nginx:latest exlli/mynginx:latest        # 给镜像打上带用户名的标签
docker push exlli/mynginx:latest        # 推送到 Docker Hub
docker pull exlli/mynginx:latest        # 从 Docker Hub 拉取

*进入容器后查看目录*
pwd          # 显示当前路径
ls           # 列出当前目录
ls /         # 看根目录
cd /usr/share/nginx/html
pwd
ls
